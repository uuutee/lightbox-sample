Magnific Popup

documents

  • Magnific Popup: Responsive jQuery Lightbox Plugin

memo

pros

  • グルーピングは親要素のclassを指定する
  • googlemapはiframeのsrcを指定する

cons

  • windowをはみでた画像はリサイズされず、スクロールする
  • 全体的に設定が複雑なので、動きごとに個別のjs指定が必要
  • アニメーションがしょぼい
  • inlineがなぜか左寄り
  • inlineはcloseボタンを設置しないと閉じられない

normal

$('.lightbox').magnificPopup({
	type: 'image',
	closeOnContentClick: true,
	image: {
		verticalFit: true
	}
});

caption

$('.lightbox-caption').magnificPopup({
	type: 'image',
	closeOnContentClick: true,
	image: {
		verticalFit: true
	}
});

gallery

$('#gallery .row').magnificPopup({
	delegate: 'a',
	type: 'image',
	tLoading: 'Loading image #%curr%...',
	gallery: {
		enabled: true,
		navigateByImgClick: true,
		preload: [0,1] // Will preload 0 - before current, and 1 after the current image
	},
	image: {
		tError: 'The image #%curr% could not be loaded.',
		// titleに一括で追加タグ指定可能
		// titleSrc: function(item) {
		// 	return item.el.attr('title') + 'by Marsel Van Oosten';
		// }
	}
});

youtube

$('.lightbox-youtube').magnificPopup({
	disableOn: 700,
	type: 'iframe',
	removalDelay: 160,
	preloader: false,
	fixedContentPos: true // trueにしないと下にずれる
});

$('.lightbox-inline').magnificPopup({
	type: 'inline',
	preloader: false,
	modal: true
});
$(document).on('click', '.modal-close', function(e) {
	e.preventDefault();
	$.magnificPopup.close();
});
